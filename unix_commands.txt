mkdir Musics (creates a folder called Musics)
ls (gives the list of the files/folders in the working directory)
cd Musics (changes the current directory to Musics)
touch song.txt (creates an empty file named song.txt)
open song.txt (Opens the file called song.txt)
rm song.txt (removes the file)
rm * (removes everything from the current directory)
echo "Hello World" > song.txt (writes "Hello World" to song.txt)
cat song.txt (displays the contents of song.txt)
mv song.txt ../ (moves song.txt to the parent directory)
cd .. (goes back to the parent directory)
rm -r Musics (removes the Musics directory and its contents)
